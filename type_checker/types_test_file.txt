let equal: type = fn(Comparable, Comparable) -> Logical; 
let not_equal: type = fn(Comparable, Comparable) -> Logical; 
let Equalable: type = equal | not_equal;

let less: type = fn(Comparable, Comparable) -> Logical; 
let greater: type = fn(Comparable, Comparable) -> Logical; 
let Comparable: type = less | greater;

let add: type = fn(Algebraic, Algebraic) -> Algebraic;
let Addable: type = add;

let sub: type = fn(Algebraic, Algebraic) -> Algebraic;
let Substractable: type = sub;

let mul: type = fn(Algebraic, Algebraic) -> Algebraic;
let Multipliable: type = mul;

let div: type = fn(Algebraic, Algebraic) -> Algebraic;
let Diviable: type = div;

let Algebraic: type = Addable | Substractable | Multipliable | Diviable;

let and: type = fn(Logical, Logical) -> Logical;
let or: type = fn(Logical, Logical) -> Logical;
let not: type = fn(Logical) -> Logical;
let Logical: type = and | or | not;
