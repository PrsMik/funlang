let fibTail: fn(int, int, int) -> int = fn(n, a, b) {
    return if (n == 0) { 
        return a; 
    } else {
        return if (n == 1) {
            return b;
        } else {
            let res: int = puts(b);
            return fibTail(n - 1, b, a + b); 
        };
    };
};

return fibTail(100, 0, 1);
